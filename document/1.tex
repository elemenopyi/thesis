\subsection*{\emph{a. Arithmatic by computer}} 
\vspace{10pt}

\begin{enumerate}
  \item[1.] 6-5+1 → (6-5)+1 \par
  \item[2.] 4-3*5 → 4-(3*5) \par
  \item[3.] 6/3*4 → (6/3)*4 \par
  \item[4.] 4*6/3 → (4*6)/3 \par
  \item[5.] 2\power3\power4 → 2\power(3\power4) \par
  \item[6.] Praat calculator results: \par
  34+5e20-5e20 = 0 \newline
  34+5e18-5e18 = 0 \newline
  34+5e16-5e16 = 32 \par
  This is because the calculator considers a number like 34 to be small enough compared to 5e18 or 5e20 to be approximated to 0. For 5e16, it results with the closest factor (3 numbers before odd factors, 4 numbers before even factors) of 8.
\end{enumerate}
